local HttpService = game:GetService("HttpService")
local TeleportService = game:GetService("TeleportService")
local Players = game:GetService("Players")

local HopTab = Window:MakeTab({Name = "🌍 AutoServerHop", Icon = "rbxassetid://4483345998", PremiumOnly = false})

HopTab:AddToggle({
    Name = "Bật AutoServerHop",
    Default = false,
    Callback = function(Value) getgenv().BF.AutoServerHop = Value end
})

function HopServer()
    local url = "https://games.roblox.com/v1/games/"..game.PlaceId.."/servers/Public?sortOrder=Asc&limit=100"
    local servers = HttpService:JSONDecode(game:HttpGet(url))
    for _,srv in pairs(servers.data) do
        if srv.playing < srv.maxPlayers then
            TeleportService:TeleportToPlaceInstance(game.PlaceId, srv.id, Players.LocalPlayer)
            break
        end
    end
end

task.spawn(function()
    while task.wait(10) do
        if getgenv().BF.AutoServerHop then
            local isFullMoon = (workspace:FindFirstChild("Moon") and workspace.Moon.Transparency == 0)
            local hasMirage = workspace:FindFirstChild("Mirage Island")
            if not isFullMoon and not hasMirage then
                print("🌍 Không có Mirage/FullMoon → Đang hop server...")
                HopServer()
            end
        end
    end
end)
